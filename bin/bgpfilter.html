 23def get_collectors():
 24    """Query the BGPStream broker and identify the collectors that are available"""
 25    data = json.load(urllib.request.urlopen(COLLECTORS_URL))
 26    result = dict((x, []) for x in PROJECTS)
 27    for coll in data["data"]["collectors"]:
 28        p = data["data"]["collectors"][coll]["project"]
 29        if p in PROJECTS:
 30            result[p].append(coll)
 31    return result

 36COLLECTORS_URL = "https://broker.bgpstream.caida.org/v2/meta/collectors"
 37COLLECTORS = get_collectors()

138    def record_mode(self, isRecord, start, end):
139        """
140        Define if retrieve from an interval or live stream
141            start and end won't be modified if isRecord is False
142
143        Args:
144            - isRecord (bool) : record or live mode
145            - start (string): Beginning of the interval.
146
147                Timestamp format : YYYY-MM-DD hh:mm:ss -> Example: 2022-01-01 10:00:00
148
149            - end (string): End of the interval.
150
151                Timestamp format : YYYY-MM-DD hh:mm:ss -> Example: 2022-01-01 10:10:00
152
153        Raises:
154            - ValueError: Start > End
155            - ValueError: Start or end not defined
156            - ValueError: Invalid date format
157        """
158        self.__isRecord = isRecord
159        if isRecord:
160            if start is None or end is None:
161                raise ValueError(
162                    "Record mode requires the from_time and until_time arguments"
163                )
164            try:
165                st = datetime.strptime(start, "%Y-%m-%d %H:%M:%S")
166                en = datetime.strptime(end, "%Y-%m-%d %H:%M:%S")
167            except ValueError:
168                raise ValueError(
169                    "Invalid record mode date format. Must be %Y-%m-%d %H:%M:%S"
170                )
171            if st > en:
172                raise ValueError(
173                    "Invalid record mode interval. Beginning must precede the end"
174                )
175
176            self.__start_time = start
177            self.__end_time = end
178        else:
179            self.__start_time = int(time())
180            self.__end_time = 0

224    @prefix_filter.setter
225    def prefix_filter(self, values: Tuple[List[str], str]):
226        """
227        Prefix filter option
228            Keep records that match to one of specified prefixes (cidr format).
229
230        Args:
231            prefix_list (List[string]):  Format: ip/subnet | Example: 130.0.192.0/21
232            match_type (string): Type of match
233                exact: Exact match
234                less: Exact match or less specific (contained by)
235                more: Exact match or more specific (contains)
236                any: less and more
237        """
238        try:
239            cidr_list, match_type = values
240        except ValueError:
241            raise ValueError(
242                "match type and prefixes list are required for filtering by prefixes"
243            )
244
245        if cidr_list is not None:
246            if len(cidr_list) == 0:
247                # "Please specify one or more prefixes when filtering by prefix"
248                return
249            if match_type not in ["exact", "less", "more", "any"]:
250                raise ValueError(
251                    "Match type must be specified and one of ['exact', 'less', 'more',"
252                    " 'any']"
253                )
254            for c in cidr_list:
255                ipaddress.ip_network(c)
256            self.__prefix_match_type_filter = match_type
257            self.__prefix_filter = cidr_list

262        Filter using specified country.
263            Keep records that the origin of their prefix is contained in country_list.

266            country_list (List[str]): List of country codes

276    @asn_filter.setter
277    def asn_filter(self, asn_list):
278        """Filter using specified AS number list
279            Skip a record if its source-AS is not one of specified AS numbers
280            Use _ symbol for negation
281
282        Args:
283            asn_list (list): List of AS numbers
284        """
285        self.__asn_filter = ""
286        f_list = []
287        not_f_list = []
288        if asn_list is not None and len(asn_list) >= 1:
289            self.__asn_list = []
290            for i in asn_list:
291                if re.match("_[0-9]+", i):
292                    not_f_list.append(i.replace("_", ""))
293                    self.__asn_list.append(i)
294                elif re.match("[0-9]+", i):
295                    f_list.append(i)
296                    self.__asn_list.append(i)
297
298            if len(f_list) >= 1:
299                self.__asn_filter += " and path (_" + "|_".join(f_list) + ")$"
300            if len(not_f_list) >= 1:
301                self.__asn_filter = " and path !(_" + "|_".join(not_f_list) + ")$"

338    def __country_by_prefix(self, p):
339        """
340        Parameters:
341            p (prefix): CIDR format. Example: 130.0.192.0/21
342
343        Returns:
344            string: country code of the given prefix.
345                None if not found in GeoOpen database
346        """
347        if p is None:
348            return None
349        else:
350            r = self.__f_country.get(p.split("/", 1)[0])
351            return r["country"]["iso_code"] if r is not None else None
352
353    def __check_country(self, e):
354        """
355        Args:
356            e (bgp record)
357
358
359        Returns:
360            boolean: if e.country code is in self.__countries_filter list
361        """
362        return not (
363            self.__countries_filter is not None
364            and e.country_code not in self.__countries_filter
365        )

383            from_time=self.start_time,
384            until_time=self.end_time,
385            data_interface=(
386                "broker"

388                else "singlefile"
389            ),
390            record_type="updates",
391            filter="elemtype announcements withdrawals"
392            + self.__asn_filter
393            + self.__ipversion,
394        )
395
396        if self.__prefix_match_type_filter is not None:
397            self._stream._maybe_add_filter(
398                "prefix-" + self.__prefix_match_type_filter, None, self.__prefix_filter
399            )
400
401        print("Starting")
402
403        if self.__data_source["source_type"] != "broker":
404            self._stream.set_data_interface_option(

419            self._stream._maybe_add_filter("collector", None, self.__collectors)
420
421        if self.__isRecord:
422            self._stream.stream.set_live_mode()
423        '''

437            if cpt % 10000 == 0:
438                nt = time()

440                ot = nt

139    def record_mode(self, isRecord, start, end):
140        """
141        Define if retrieve from an interval or live stream
142            start and end won't be modified if isRecord is False
143
144        Args:
145            - isRecord (bool) : record or live mode
146            - start (string): Beginning of the interval.
147
148                Timestamp format : YYYY-MM-DD hh:mm:ss -> Example: 2022-01-01 10:00:00
149
150            - end (string): End of the interval.
151
152                Timestamp format : YYYY-MM-DD hh:mm:ss -> Example: 2022-01-01 10:10:00
153
154        Raises:
155            - ValueError: Start > End
156            - ValueError: Start or end not defined
157            - ValueError: Invalid date format
158        """
159        self.__isRecord = isRecord
160        if isRecord:
161            if start is None or end is None:
162                raise ValueError(
163                    "Record mode requires the from_time and until_time arguments"
164                )
165            try:
166                st = datetime.strptime(start, "%Y-%m-%d %H:%M:%S")
167                en = datetime.strptime(end, "%Y-%m-%d %H:%M:%S")
168            except ValueError:
169                raise ValueError(
170                    "Invalid record mode date format. Must be %Y-%m-%d %H:%M:%S"
171                )
172            if st > en:
173                raise ValueError(
174                    "Invalid record mode interval. Beginning must precede the end"
175                )
176
177            self.__start_time = start
178            self.__end_time = end
179        else:
180            self.__start_time = int(time())
181            self.__end_time = 0

185        """
